# F2.2 Quản Lý Khách

## Tổng quan

Host quản lý danh sách khách mời: thêm tay từng người, import bulk từ Excel, xóa/sửa, xem trạng thái RSVP. Mỗi khách được cấp `token` unique để tạo link thiệp cá nhân. Có thể tạo link form tự đăng ký để khách điền tên vào.

## Controller

```php
// app/Http/Controllers/Dashboard/GuestController.php
class GuestController extends Controller
{
    public function __construct(private GuestService $guestService) {}

    public function index(Event $event): Response
    {
        $this->authorize('update', $event);

        return Inertia::render('Dashboard/Events/Guests', [
            'event'  => $event->only('id', 'slug', 'title'),
            'guests' => Guest::where('event_id', $event->id)
                ->with('rsvp:guest_id,status,plus_one')
                ->orderBy('name')
                ->paginate(50),
            'stats'  => $this->guestService->getStats($event),
            'selfRegisterUrl' => route('invitation.register.form', $event->slug),
        ]);
    }

    public function store(StoreGuestRequest $request, Event $event): JsonResponse
    {
        $this->authorize('update', $event);

        $guest = $this->guestService->addGuest($event, $request->validated());

        return response()->json($guest->only('id', 'name', 'email', 'phone', 'table_no', 'token'), 201);
    }

    public function update(UpdateGuestRequest $request, Event $event, Guest $guest): JsonResponse
    {
        $this->authorize('update', $event);
        abort_if($guest->event_id !== $event->id, 403);

        $guest->update($request->validated());

        return response()->json($guest->fresh()->only('id', 'name', 'email', 'phone', 'table_no'));
    }

    public function destroy(Event $event, Guest $guest): JsonResponse
    {
        $this->authorize('update', $event);
        abort_if($guest->event_id !== $event->id, 403);

        $guest->delete();   // cascade xóa rsvp, wishes liên quan

        return response()->json(['deleted' => true]);
    }

    public function import(ImportGuestRequest $request, Event $event): JsonResponse
    {
        $this->authorize('update', $event);

        $result = $this->guestService->importExcel($event, $request->file('file'));

        return response()->json($result);
    }

    public function export(Event $event): BinaryFileResponse
    {
        $this->authorize('update', $event);

        return Excel::download(
            new GuestsExport($event),
            "khach-moi-{$event->slug}.xlsx"
        );
    }
}
```

```php
// app/Services/GuestService.php
class GuestService
{
    public function __construct(private GuestRepository $guestRepo) {}

    public function addGuest(Event $event, array $data): Guest
    {
        return $this->guestRepo->create([
            'event_id' => $event->id,
            'name'     => $data['name'],
            'email'    => $data['email']    ?? null,
            'phone'    => $data['phone']    ?? null,
            'table_no' => $data['table_no'] ?? null,
            'token'    => Str::random(32),
            'source'   => 'manual',
        ]);
    }

    public function importExcel(Event $event, UploadedFile $file): array
    {
        $import = new GuestsImport($event);

        Excel::import($import, $file);

        return [
            'imported' => $import->importedCount,
            'skipped'  => $import->skippedCount,
            'errors'   => $import->errors,   // [{row: 3, message: "Email không hợp lệ"}]
        ];
    }

    public function getStats(Event $event): array
    {
        $total     = Guest::where('event_id', $event->id)->count();
        $withEmail = Guest::where('event_id', $event->id)->whereNotNull('email')->count();
        $rsvpYes   = Rsvp::where('event_id', $event->id)->where('status', 'yes')->count();
        $rsvpNo    = Rsvp::where('event_id', $event->id)->where('status', 'no')->count();
        $rsvpMaybe = Rsvp::where('event_id', $event->id)->where('status', 'maybe')->count();
        $noRsvp    = $total - ($rsvpYes + $rsvpNo + $rsvpMaybe);

        return compact('total', 'withEmail', 'rsvpYes', 'rsvpNo', 'rsvpMaybe', 'noRsvp');
    }
}
```

## Excel Import (maatwebsite/excel)

```php
// app/Imports/GuestsImport.php
class GuestsImport implements ToCollection, WithHeadingRow, SkipsOnFailure
{
    use Importable, SkipsFailures;

    public int   $importedCount = 0;
    public int   $skippedCount  = 0;
    public array $errors        = [];

    public function __construct(public Event $event) {}

    public function collection(Collection $rows): void
    {
        foreach ($rows as $index => $row) {
            $rowNum = $index + 2;   // +2 vì row 1 là heading

            // Validate
            $validator = Validator::make($row->toArray(), [
                'ten'   => ['required', 'string', 'max:100'],
                'email' => ['nullable', 'email', 'max:150'],
                'sdt'   => ['nullable', 'string', 'max:20'],
                'ban'   => ['nullable', 'string', 'max:20'],
            ]);

            if ($validator->fails()) {
                $this->errors[]   = ['row' => $rowNum, 'message' => implode(', ', $validator->errors()->all())];
                $this->skippedCount++;
                continue;
            }

            Guest::create([
                'event_id' => $this->event->id,
                'name'     => $row['ten'],
                'email'    => $row['email'] ?? null,
                'phone'    => $row['sdt']   ?? null,
                'table_no' => $row['ban']   ?? null,
                'token'    => Str::random(32),
                'source'   => 'import',
            ]);

            $this->importedCount++;
        }
    }
}
```

```php
// app/Exports/GuestsExport.php
class GuestsExport implements FromCollection, WithHeadings
{
    public function __construct(private Event $event) {}

    public function collection(): Collection
    {
        return Guest::where('event_id', $this->event->id)
            ->with('rsvp:guest_id,status,plus_one')
            ->get()
            ->map(fn ($g) => [
                $g->name,
                $g->email,
                $g->phone,
                $g->table_no,
                $g->rsvp?->status ?? 'chưa xác nhận',
                $g->rsvp?->plus_one ?? 0,
                route('invitation.show', $this->event->slug) . '?t=' . $g->token,
            ]);
    }

    public function headings(): array
    {
        return ['Tên', 'Email', 'SĐT', 'Bàn', 'RSVP', 'Người đi cùng', 'Link thiệp cá nhân'];
    }
}
```

## Self-Register Flow (Khách tự điền tên)

```php
// app/Http/Controllers/SelfRegisterController.php
class SelfRegisterController extends Controller
{
    public function form(string $slug): View
    {
        $event = Event::where('slug', $slug)->where('status', 'published')->firstOrFail();
        abort_if(!$event->self_register_enabled, 404);

        return view('invitation.register', compact('event'));
    }

    public function store(Request $request, string $slug): RedirectResponse
    {
        $event = Event::where('slug', $slug)->where('status', 'published')->firstOrFail();
        abort_if(!$event->self_register_enabled, 403);

        $data = $request->validate([
            'name'  => ['required', 'string', 'max:100'],
            'email' => ['nullable', 'email', 'max:150'],
            'phone' => ['nullable', 'string', 'max:20'],
        ]);

        // Rate limit: 10 lần / 10 phút per IP
        RateLimiter::attempt(
            "self-register:{$request->ip()}",
            10,
            fn () => null,
        ) || abort(429);

        Guest::create([
            'event_id' => $event->id,
            'name'     => $data['name'],
            'email'    => $data['email'] ?? null,
            'phone'    => $data['phone'] ?? null,
            'token'    => Str::random(32),
            'source'   => 'self_register',
        ]);

        return redirect()->route('invitation.register.success', $slug);
    }

    public function success(string $slug): View
    {
        $event = Event::where('slug', $slug)->where('status', 'published')->firstOrFail();
        return view('invitation.register-success', compact('event'));
    }
}
```

## Import Error Report (Vue component con)

```vue
<!-- resources/js/Components/ImportExcelPanel.vue -->
<script setup>
import { ref } from 'vue'
import axios from 'axios'

const props = defineProps({ event: Object })
const emit  = defineEmits(['close'])

const file    = ref(null)
const result  = ref(null)
const loading = ref(false)

async function submit() {
    if (!file.value) return
    loading.value = true

    const form = new FormData()
    form.append('file', file.value)

    const res = await axios.post(
        route('dashboard.events.guests.import', props.event.slug),
        form,
        { headers: { 'Content-Type': 'multipart/form-data' } }
    )
    result.value  = res.data
    loading.value = false
}
</script>

<template>
    <div class="bg-yellow-50 border border-yellow-200 rounded p-4 mb-4">
        <div class="flex items-center gap-3">
            <input type="file" accept=".xlsx,.xls,.csv" @change="e => file = e.target.files[0]">
            <button @click="submit" :disabled="!file || loading" class="btn-primary">
                {{ loading ? 'Đang import...' : 'Import' }}
            </button>
            <a href="/templates/mau-import-khach.xlsx" class="text-sm text-blue-600 underline">
                Tải file mẫu
            </a>
            <button @click="emit('close')" class="ml-auto text-gray-400">✕</button>
        </div>

        <!-- Kết quả -->
        <div v-if="result" class="mt-3">
            <p class="text-sm">

                    Bỏ qua {{ result.skipped }} dòng lỗi.
```

## Edge Cases

| Tình huống | Xử lý |
|-----------|-------|
| Import Excel thiếu cột `ten` | Row bị skip, ghi vào `errors[]` với số dòng |
| Import email trùng trong cùng event | Cho phép — 2 khách khác nhau có thể dùng cùng email |
| Import 1,000+ rows | `collection()` xử lý từng dòng, không load all vào RAM |
| `table_no` là số hay text | `VARCHAR(20)` — không ép kiểu, "Bàn 1" hay "1" đều OK |
| Guest bị xóa còn wishes/rsvp | `ON DELETE CASCADE` trong migration |
| `self_register_enabled = false` | Form → 404, POST → 403 |
| Self-register spam | `RateLimiter` 10 lần / 10 phút per IP |
| Copy link cá nhân khi không có HTTPS | `navigator.clipboard` chỉ hoạt động trên HTTPS/localhost |
| Export file tên có dấu | `khach-moi-{slug}.xlsx` — slug đã là ASCII |
| Guest thuộc event khác bị update/delete | `abort_if($guest->event_id !== $event->id, 403)` |
