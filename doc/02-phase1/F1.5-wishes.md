# F1.5 Lời Chúc

## Tổng quan

Khách gửi lời chúc từ trang thiệp public (Blade). Lời chúc hiển thị trực tiếp trên thiệp theo thời gian thực (polling). Host có thể pin/ẩn từng lời chúc trong dashboard.

---

## Flow

```
Trang thiệp /thiep/{slug}
  → Section "Sổ lời chúc"
    → Form submit (POST)
      → Hiển thị ngay trong danh sách (optimistic hoặc reload section)
      → Notification push cho host
```

## Public Controller

```php
// app/Http/Controllers/WishController.php
class WishController extends Controller
{
    public function __construct(private WishService $wishService) {}

    public function store(StoreWishRequest $request, string $slug): JsonResponse|RedirectResponse
    {
        $event = Event::where('slug', $slug)->where('status', 'published')->firstOrFail();

        abort_if(!$event->wishes_enabled, 403);

        $wish = $this->wishService->submit(
            event:   $event,
            data:    $request->validated(),
            token:   $request->query('t'),
            ip:      $request->ip(),
        );

        if ($request->expectsJson()) {
            return response()->json([
                'id'         => $wish->id,
                'name'       => $wish->guest->name ?? $request->validated('name'),
                'message'    => $wish->message,
                'created_at' => $wish->created_at->diffForHumans(),
            ], 201);
        }

        return back()->with('wish_sent', true);
    }

    public function index(string $slug, Request $request): JsonResponse
    {
        $event = Event::where('slug', $slug)->where('status', 'published')->firstOrFail();

        $wishes = $this->wishService->getPublicWishes($event, $request->query('after'));

        return response()->json($wishes);
    }
}
```

// app/Http/Requests/StoreWishRequest.php
class StoreWishRequest extends FormRequest
{
    public function rules(): array
    {
        // Token mode: name optional (lấy từ guest)
        $nameRule = $this->query('t')
            ? ['nullable', 'string', 'max:100']
            : ['required', 'string', 'max:100'];

        return [
            'name'    => $nameRule,
            'message' => ['required', 'string', 'min:2', 'max:500'],
        ];
    }
}

## Service

```php
// app/Services/WishService.php
class WishService
{
    public function __construct(
        private WishRepository $wishRepo,
        private NotificationService $notifService,
    ) {}

    public function submit(Event $event, array $data, ?string $token, string $ip): Wish
    {
        $guest = $token ? Guest::where('token', $token)->first() : null;

        // Anonymous: tạo guest tạm nếu chưa có
        if (!$guest) {
            $guest = Guest::firstOrCreate(
                ['event_id' => $event->id, 'name' => $data['name'], 'source' => 'self_register'],
                ['token' => Str::random(32)],
            );
        }

        $wish = $this->wishRepo->create([
            'event_id'  => $event->id,
            'guest_id'  => $guest->id,
            'message'   => $data['message'],
            'is_pinned' => false,
            'is_hidden' => false,
        ]);

        $this->notifService->notifyHost($event, 'wish', [
            'guest_name' => $guest->name,
            'wish_id'    => $wish->id,
            'preview'    => Str::limit($wish->message, 60),
            'event_id'   => $event->id,
        ]);

        return $wish;
    }

    public function getPublicWishes(Event $event, ?string $afterId): array
    {
        $query = Wish::where('event_id', $event->id)
            ->where('is_hidden', false)
            ->with('guest:id,name')
            ->orderByDesc('is_pinned')   // pinned lên đầu
            ->orderByDesc('created_at');

        // Polling incremental: chỉ lấy wishes mới hơn last seen id
        if ($afterId) {
            $query->where('id', '>', (int) $afterId);
        } else {
            $query->limit(50);
        }

        return $query->get()->map(fn ($w) => [
            'id'         => $w->id,
            'name'       => $w->guest->name ?? 'Ẩn danh',
            'message'    => $w->message,
            'is_pinned'  => $w->is_pinned,
            'created_at' => $w->created_at->diffForHumans(),
        ])->toArray();
    }
}
```

## Dashboard Controller

```php
// app/Http/Controllers/Dashboard/DashboardWishController.php
class DashboardWishController extends Controller
{
    public function __construct(private WishService $wishService) {}

    public function index(Event $event): Response
    {
        $this->authorize('update', $event);

        return Inertia::render('Dashboard/Events/Wishes', [
            'event'  => $event->only('id', 'slug', 'title'),
            'wishes' => Wish::where('event_id', $event->id)
                ->with('guest:id,name,email')
                ->orderByDesc('is_pinned')
                ->orderByDesc('created_at')
                ->paginate(30),
        ]);
    }

    public function pin(Event $event, Wish $wish): JsonResponse
    {
        $this->authorize('update', $event);
        abort_if($wish->event_id !== $event->id, 403);

        $wish->update(['is_pinned' => !$wish->is_pinned]);

        return response()->json(['is_pinned' => $wish->is_pinned]);
    }

    public function hide(Event $event, Wish $wish): JsonResponse
    {
        $this->authorize('update', $event);
        abort_if($wish->event_id !== $event->id, 403);

        $wish->update(['is_hidden' => !$wish->is_hidden]);

        return response()->json(['is_hidden' => $wish->is_hidden]);
    }

    public function destroy(Event $event, Wish $wish): JsonResponse
    {
        $this->authorize('update', $event);
        abort_if($wish->event_id !== $event->id, 403);

        $wish->delete();

        return response()->json(['deleted' => true]);
    }
}
```

## Hiển thị trên thiệp (Blade + JS Polling)

```blade
{{-- resources/views/partials/wishes-section.blade.php --}}
<section id="wishes-section" class="wishes-section"
         data-event-slug="{{ $event->slug }}"
         data-wishes-enabled="{{ $event->wishes_enabled ? 'true' : 'false' }}">

    <h2>{{ __('wishes.section_title') }}</h2>

    {{-- Form gửi lời chúc --}}
    @if ($event->wishes_enabled)
    <form id="wish-form" class="wish-form">
        @csrf
        @if (!$guest)
            <input type="text" name="name" placeholder="{{ __('wishes.your_name') }}" maxlength="100" required>
        @else
            <input type="hidden" name="name" value="{{ $guest->name }}">
            <p>{{ __('wishes.greeting', ['name' => $guest->name]) }}</p>
        @endif
        <textarea name="message" placeholder="{{ __('wishes.placeholder') }}" maxlength="500" required></textarea>
        <button type="submit">{{ __('wishes.send') }}</button>
        <span class="wish-counter">0/500</span>
    </form>
    @endif

    {{-- Danh sách lời chúc --}}
    <div id="wishes-list" class="wishes-list">
        <div class="wishes-loading">{{ __('wishes.loading') }}</div>
    </div>
</section>

<script>
(function () {
    const slug         = document.querySelector('#wishes-section').dataset.eventSlug;
    const wishesEnabled = document.querySelector('#wishes-section').dataset.wishesEnabled === 'true';
    const list         = document.getElementById('wishes-list');
    const form         = document.getElementById('wish-form');
    let   lastId       = 0;

    // Render helper
    function renderWish(w) {
        const el = document.createElement('div');
        el.className = 'wish-item' + (w.is_pinned ? ' wish-pinned' : '');
        el.dataset.id = w.id;
        el.innerHTML = `
            <strong>${escapeHtml(w.name)}</strong>
            <p>${escapeHtml(w.message)}</p>
            <time>${w.created_at}</time>
        `;
        return el;
    }

    function escapeHtml(str) {
        return str.replace(/[&<>"']/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m]));
    }

    // Initial load
    async function loadWishes() {
        const res  = await fetch(`/thiep/${slug}/wishes`);
        const data = await res.json();
        list.innerHTML = '';
        data.forEach(w => {
            list.appendChild(renderWish(w));
            if (w.id > lastId) lastId = w.id;
        });
    }

    // Polling mỗi 15s — chỉ lấy wishes mới hơn lastId
    async function pollWishes() {
        if (!lastId) return;
        const res  = await fetch(`/thiep/${slug}/wishes?after=${lastId}`);
        const data = await res.json();
        data.forEach(w => {
            if (!document.querySelector(`[data-id="${w.id}"]`)) {
                list.prepend(renderWish(w));   // prepend = mới nhất lên đầu
                if (w.id > lastId) lastId = w.id;
            }
        });
    }

    // Form submit
    if (wishesEnabled && form) {
        // Character counter
        const textarea = form.querySelector('textarea[name="message"]');
        const counter  = form.querySelector('.wish-counter');
        textarea.addEventListener('input', () => {
            counter.textContent = `${textarea.value.length}/500`;
        });

        const token = document.querySelector('meta[name="csrf-token"]')?.content;
        const tParam = new URLSearchParams(location.search).get('t');
        const actionUrl = `/thiep/${slug}/wishes` + (tParam ? `?t=${tParam}` : '');

        form.addEventListener('submit', async (e) => {
            e.preventDefault();
            const btn = form.querySelector('button[type="submit"]');
            btn.disabled = true;

            const body = new FormData(form);
            const res  = await fetch(actionUrl, {
                method: 'POST',
                headers: { 'X-CSRF-TOKEN': token, 'Accept': 'application/json' },
                body,
            });

            if (res.ok) {
                const wish = await res.json();
                list.prepend(renderWish(wish));
                lastId = Math.max(lastId, wish.id);
                form.reset();
                counter.textContent = '0/500';
            }

            btn.disabled = false;
        });
    }

    // Khởi động
    loadWishes().then(() => {
        setInterval(pollWishes, 15000);
    });
})();
</script>
```

## Edge Cases

| Tình huống | Xử lý |
|-----------|-------|
| `wishes_enabled = false` | POST → 403. Form không render trên Blade |
| Spam > 5 lần / 10 phút | `throttle:wishes` → 429 |
| XSS trong message/name | `escapeHtml()` phía JS. Laravel auto-escape trong Blade `{{ }}` |
| Message 501 ký tự | Validation `max:500` fail → lỗi field |
| Token sai → anonymous | `resolveGuest()` null → `firstOrCreate` guest mới |
| Guest gửi 2 lần cùng tên + message | Tạo 2 record riêng (không dedup — lời chúc có thể trùng nội dung) |
| Wish thuộc event khác bị pin/hide | `abort_if($wish->event_id !== $event->id, 403)` |
| Poll khi không có wish mới | API trả `[]`, không re-render |
| Host xóa wish đang hiển thị | Khách thấy đến lần reload tiếp, không real-time xóa phía client |
